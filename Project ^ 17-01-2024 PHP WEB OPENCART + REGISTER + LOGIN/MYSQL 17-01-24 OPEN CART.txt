CREATE TABLE products (
product_id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(100) UNIQUE NOT NULL,
    price DOUBLE(20,2) NOT NULL,
    url VARCHAR (200) NOT NULL
); 

CREATE TABLE slider (
slider_id INT PRIMARY KEY AUTO_INCREMENT,
   url VARCHAR (400) UNIQUE NOT NULL,
    product_id INT NOT NULL,
    FOREIGN KEY (product_id)REFERENCES products(product_id)
); 


CREATE TABLE marketplace_p (
marketplace_id INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(100) NOT NULL UNIQUE, 
    url VARCHAR (400) NOT NULL UNIQUE
);

CREATE TABLE slider_m (
sliderm_id INT PRIMARY KEY AUTO_INCREMENT,

    url VARCHAR(400) NOT NULL UNIQUE,
    marketplace_id INT NOT NULL,
    FOREIGN KEY (marketplace_id) REFERENCES marketplace_p(marketplace_id)
    
);



CREATE TABLE register (
id iNT PRIMARY KEY AUTO_INCREMENT,
  login VARCHAR(200) NOT NULL UNIQUE,
   password VARCHAR(200) NOT NULL UNIQUE,
    email VARCHAR(200) NOT NULL UNIQUE,
    first_name VARCHAR(100) NOT NULL UNIQUE,
    last_name VARCHAR (100) NOT NULL UNIQUE
);




CREATE TABLE bank_cart (
cart_id INT PRIMARY KEY AUTO_INCREMENT,
  number VARCHAR(150) UNIQUE NOT NULL,
  expir VARCHAR(120) UNIQUE NOT NULL,
  cvv VARCHAR (39) UNIQUE NOT NULL
);
